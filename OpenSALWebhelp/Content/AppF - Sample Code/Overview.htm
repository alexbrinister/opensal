SAL Sample Code

The following examples are more extensive than those provided elsewhere in this document.

The code in this example creates two benchmark programs:

  • bm_fftm_loop_7447.ppc for the PowerPC 7447
  • bm_fftm_loop_7400.ppc for the PowerPC 7400

Using strip-mining to maximize the reuse of the data while it is resident in the L1 cache, each of these benchmarks performs the following tasks:

  1. Transposes the 16 by 1K data set to be 1K by 16.
  2. Converts 16K complex short elements into 16K complex split float elements.
  3. Performs 1,024 16-point FFTs on the complex split float data.
  4. Sums the real part of the 16K FFT output points.
  5. Sums the imaginary part of the 16K FFT output points.
  6. Computes the magnitude square of the FFT output.
  7. Sums the 16K output magnitude square output elements.
  8. Multiplies the 16K FFT output points by a 16K vector of floats.

Benchmark

bm_fftm_loop.c

#include <stdio.h>
#include <assert.h>
#include <stdlib.h>
#include <sal.h>
#include <mcos.h>
#include "mem_utils.h"
#define FFT_LOG2N ( 4 ) /* 16-point FFT */
#define NFFT_ELEMS ( 1 << FFT_LOG2N )
#define NFFTS ( 1024 ) /* 1024, 16-point FFTs */
#define VECTOR_LEN ( ( 1 << FFT_LOG2N ) * NFFTS )
#define NFFTS_PER_STRIP ( 32 ) /* Call fftm with 32, 16-point FFTs */
#define NELEMS_PER_STRIP ( 48 ) /* 48 element in each vector SAL call */
#define NELEMS_OTHER_INPUTS ( 48 )
#define FLUSH_NBYTES ( 8 * 1024 * 1024 ) /* Big flush buffer */
#define NLOOPS ( 10 ) /* Average time of N loops is reported */
SAL_i32 main ( SAL_i32 argc, SAL_char * argv [ ] )
{
  SAL_i32 n;
  SAL_i32 n_elems_left,
  n_elems_call,
  n_elems_orig;
  SAL_i32 n_ffts_left,
  n_ffts_call,
  n_ffts_orig;
  SAL_ui32 n_bytes;
  SAL_ui16 * pu16_src;
  SMB_handle h_smb_u16_src;
  SAL_zf32 zf32_A;
  SMB_handle h_smb_zf32_Ar;
  SMB_handle h_smb_zf32_Ai;
  SAL_zf32 zf32_T;
  SMB_handle h_smb_zf32_Tr;
  SMB_handle h_smb_zf32_Ti;
  SAL_f32 * pf32_B;
  SMB_handle h_smb_f32_B;
  SAL_f32 f32_C;
  SAL_f32 * pf32_D;
  SMB_handle h_smb_f32_D;
  SAL_zf32 zf32_E;
  SMB_handle h_smb_zf32_Er;
  SMB_handle h_smb_zf32_Ei;
  SAL_cf32 cf32_F;
  SAL_ui8 * pu8_flush;
  SMB_handle h_smb_u8_flush;
  FFT_setup h_FFT;
  TMR_ts p_tm_beg [ NLOOPS ];
  TMR_ts p_tm_end [ NLOOPS ];
  TMR_timespec time_elapsed;
  SAL_f64 call_time;
  SAL_f64 total_time;
  /* ---- Allocate the input vector of complex shorts ---- */
  n_bytes = sizeof ( SAL_ui16 ) * VECTOR_LEN;
  pu16_src = ( SAL_ui16 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_u16_src );
  assert ( pu16_src != NULL );
  /* ---- Allocate the input/output A vector ---- */
  n_bytes = sizeof ( SAL_f32 ) * VECTOR_LEN;
  zf32_A.realp = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_zf32_Ar );
  assert ( zf32_A.realp != NULL );
  zf32_A.imagp = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_zf32_Ai );
  assert ( zf32_A.imagp != NULL );
  /* ---- Allocate the input/output A vector ---- */
  n_bytes = sizeof ( SAL_f32 ) * VECTOR_LEN;
  zf32_T.realp = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_zf32_Tr );
  assert ( zf32_T.realp != NULL );
  zf32_T.imagp = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_zf32_Ti );
  assert ( zf32_T.imagp != NULL );
  /* ---- Allocate the output/input B vector ---- */
  n_bytes = sizeof ( SAL_f32 ) * NELEMS_OTHER_INPUTS;
  pf32_B = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_f32_B );
  assert ( pf32_B != NULL );
  /* ---- Allocate the output/input D vector ---- */
  n_bytes = sizeof ( SAL_f32 ) * NELEMS_OTHER_INPUTS;
  pf32_D = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_f32_D );
assert ( pf32_D != NULL );
  /* ---- Allocate the output/input E vector ---- */
  n_bytes = sizeof ( SAL_f32 ) * VECTOR_LEN;
  zf32_E.realp = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_zf32_Er );
  assert ( zf32_E.realp != NULL );
  zf32_E.imagp = ( SAL_f32 * ) mem_alloc ( n_bytes,
MEM_FROM_SMB,
( void * ) &h_smb_zf32_Ei );
  assert ( zf32_E.imagp != NULL );
  /* ---- Allocate the "flush" buffer ---- */
  pu8_flush = ( SAL_ui8 * ) mem_alloc ( FLUSH_NBYTES,
MEM_FROM_SMB,
( void * ) &h_smb_u8_flush );
  assert ( pu8_flush != NULL );
  /* ---- Call FFT setup to allocate the FFT weights ---- */
  fft1d_setup ( FFT_LOG2N, 0, &h_FFT, &n_bytes );
  assert ( n_bytes != 0 );
  /* ---- Time the group of SAL functions ---- */
  for ( n = 0; n < NLOOPS; n++ )
  {
    /* Flush the data from the cache */
    cache_load_1d ( ( SAL_char * ) pu8_flush, FLUSH_NBYTES, 0, 0 );
    cache_flush_1d ( ( SAL_char * ) pu8_flush, FLUSH_NBYTES, 0, 0 );
    tmr_timestamp ( &( p_tm_beg [ n ] ) );
      n_ffts_left = n_ffts_orig = NFFTS;
      /* Perform the FFTs in batches */
      while ( n_ffts_left )
      {
        n_ffts_call = ( n_ffts_left >= NFFTS_PER_STRIP ) ?
        NFFTS_PER_STRIP : n_ffts_left;
      /* Treat the 16-bit real/imag pair as one 32-bit SAL_f32 */
      mtrans2x ( ( SAL_f32 * ) ( void * ) pu16_src,
        zf32_T.realp,
        n_ffts_call, /* ncols src, nrows dst */
        NFFT_ELEMS, /* nrows src, ncols dst */
        NFFTS, /* column stride src */
        NFFT_ELEMS, /* column stride dst */
        SAL_NN );
      /* Convert the short data to float */
      vconvert_cu16_zf32x ( ( SAL_cui16 * ) zf32_T.realp,
          2,
          &zf32_A,
          1,
          NULL,
          NULL,
          ( n_ffts_call * NFFT_ELEMS ),
          0,
          SAL_CC );
      /* Perform the FFTm */
      fftm_ziptx ( &h_FFT,
          &zf32_A,
          1,
          0,
          &zf32_T,
          FFT_LOG2N,
          n_ffts_call,
          FFT_FORWARD,
          SAL_CC );
      n_elems_left = n_elems_orig = ( n_ffts_call * NFFT_ELEMS );
      /* Perform the vector operations on the FFTm output */
      while ( n_elems_left )
        {
        n_elems_call = ( n_elems_left >= NELEMS_PER_STRIP )
          ? NELEMS_PER_STRIP
          : n_elems_left;
        /* Sum the real part */
        svex ( zf32_A.realp, 1, &( cf32_F.real ),
          n_elems_call, SAL_CC );
        /* Sum the imaginary parts */
        svex ( zf32_A.imagp, 1, &( cf32_F.imag ),
          n_elems_call, SAL_CC );
        /* Vector magnitude square */
        zvmagsx ( &zf32_A,
            1,
            pf32_B,
            1,
            n_elems_call,
            SAL_CC );
        /* Sum of vector elements */
        svex ( pf32_B, 1, &f32_C, n_elems_call, SAL_CC );
        /* Multiply FFT output by real vector D */
        zrvmulx ( &zf32_A,
            1,
            pf32_D,
            1,
            &zf32_E,
            1,
            n_elems_call,
            SAL_CCC );
          /* Bump to the next strip */
          zf32_A.realp += n_elems_call;
          zf32_A.imagp += n_elems_call;
          zf32_E.realp += n_elems_call;
          zf32_E.imagp += n_elems_call;
          /* Reduce the loop count */
          n_elems_left -= n_elems_call;
          } /* End: while ( n_left ) */
          /* Restore the data pointers pointers */
          zf32_A.realp -= n_elems_orig;
          zf32_A.imagp -= n_elems_orig;
          zf32_E.realp -= n_elems_orig;
          zf32_E.imagp -= n_elems_orig;
          /* Reduce the number of FFTs left to perform */
          n_ffts_left -= n_ffts_call;
          /* Bump to the next FFT strip */
          zf32_A.realp += ( n_ffts_call * NFFT_ELEMS );
          zf32_A.imagp += ( n_ffts_call * NFFT_ELEMS );
          pu16_src += ( n_ffts_call * 2 );
        } /* End: while ( n_ffts_left ) */
      /* Restore the pointers */
      zf32_A.realp -= ( n_ffts_orig * NFFT_ELEMS );
      zf32_A.imagp -= ( n_ffts_orig * NFFT_ELEMS );
      pu16_src -= ( n_ffts_orig * 2 );
      /* Stop the timer */
      tmr_timestamp ( &( p_tm_end [ n ] ) );
    } /* End: for ( n = 0; n < NLOOPS; n++ ) */
    /* ---- Compute and report the average time ---- */
    total_time = 0.0;
    for ( n = 0; n < NLOOPS; n++ )
    {
      tmr_diff ( p_tm_beg [ n ], p_tm_end [ n ], 0, &time_elapsed );
      call_time = ( ( ( SAL_f64 ) time_elapsed.tv_sec * 1E6 ) +
        ( SAL_f64 ) time_elapsed.tv_nsec / 1E3 );
    total_time += call_time;
    }
  printf ( "\nAverage time %16.2f usecs\n", total_time / NLOOPS );
  /* ---- Free the input vector of complex shorts ---- */
  mem_free ( pu16_src, MEM_FROM_SMB, ( void * ) &h_smb_u16_src );
  /* ---- Free the input/output A vector ---- */
  mem_free ( zf32_A.realp, MEM_FROM_SMB, ( void * ) &h_smb_zf32_Ar );
  mem_free ( zf32_A.imagp, MEM_FROM_SMB, ( void * ) &h_smb_zf32_Ai );
  /* ---- Free the temporary vector T ---- */
  mem_free ( zf32_T.realp, MEM_FROM_SMB, ( void * ) &h_smb_zf32_Tr );
  mem_free ( zf32_T.imagp, MEM_FROM_SMB, ( void * ) &h_smb_zf32_Ti );
  /* ---- Free the output B vector ---- */
  mem_free ( pf32_B, MEM_FROM_SMB, ( void * ) &h_smb_f32_B );
  /* ---- Free the output D vector ---- */
  mem_free ( pf32_D, MEM_FROM_SMB, ( void * ) &h_smb_f32_D );
  /* ---- Free the output E vector ---- */
  mem_free ( zf32_E.realp, MEM_FROM_SMB, ( void * ) &h_smb_zf32_Er );
  mem_free ( zf32_E.imagp, MEM_FROM_SMB, ( void * ) &h_smb_zf32_Ei );
  /* ---- Free the flush buffer ---- */
  mem_free ( pu8_flush, MEM_FROM_SMB, ( void * ) &h_smb_u8_flush );
  /* ---- Free the FFT weights ---- */
  fft_free ( &h_FFT );
  return ( 0 );
}

Makefile Example

You can use this sample makefile to build the benchmark programs.

# ---- Beginning of Makefile -------------------------- #
# ---- Targets ---------------------------------------- #
EXAMPLE = bm_fftm_loop
all: $(EXAMPLE)_7400.ppc $(EXAMPLE)_7447.ppc
# ---- Object List ------------------------------------ #
OBJS_PPC7400_BE = \
$(EXAMPLE).oppc7400 \
mem_utils.oppc7400
OBJS_PPC7447_BE = \
$(EXAMPLE).oppc7447 \
mem_utils.oppc7447
# ---- Tools ------------------------------------------ #
TOOLS_DIR = $(MC_ROOT_DIR)/bin/$(MC_HOST_DIR)
AS = $(TOOLS_DIR)/ccmc
CC = $(TOOLS_DIR)/ccmc
LD = $(TOOLS_DIR)/ccmc
# ---- Includes --------------------------------------- #
INCLUDE_PATH = -I.
# ---- CC & LD Flags ---------------------------------- #
OPTIONS = -Ot
CCFLAGS = $(OPTIONS) -w2
LDFLAGS = $(OPTIONS)
# ---- Suffixes --------------------------------------- #
.SUFFIXES: .c .oppc7400 .oppc7447 .ppc
# ---- Build Rules ------------------------------------ #
.c.oppc7400:
$(CC) -t ppc7400 $(CCFLAGS) $(INCLUDE_PATH) -o $*.oppc7400 -c <
.c.oppc7447:
$(CC) -t ppc7447 $(CCFLAGS) $(INCLUDE_PATH) -o $*.oppc7447 -c <
# ---- Build Targets ---------------------------------- #
$(EXAMPLE)_7400.ppc: $(OBJS_PPC7400_BE) Makefile
@echo ""; echo "-- Building " $@ " --"; echo " "
$(LD) -t ppc7400 $(LDFLAGS) -o $@ $(OBJS_PPC7400_BE)
$(EXAMPLE)_7447.ppc: $(OBJS_PPC7447_BE) Makefile
@echo ""; echo "-- Building " $@ " --"; echo " "
$(LD) -t ppc7447 $(LDFLAGS) -o $@ $(OBJS_PPC7447_BE)
# ---- Clean Rules ------------------------------------ #
clean:
rm -f $(OBJS_PPC7400_BE) $(EXAMPLE)_7400.ppc
rm -f $(OBJS_PPC7447_BE) $(EXAMPLE)_7447.ppc
rm -f *.map *.inf
# ---- End of Makefile -------------------------------- #

Memory Examples

mem_utils.c

#include <mcos.h>
#include <stdio.h>
#include <stdlib.h>
#include "mem_utils.h"
/** -- Allocate memory with an SMB or malloc() -- **/
void * mem_alloc ( SAL_ui32 nbytes, SAL_i32 flag, void *ph_mem_info )
{
#if defined ( _MC_EXEC )
  SAL_i32 rc;
  SMB_mapping_handle hmap;
  SMB_mapping_info hmap_info;
  if ( flag == MEM_FROM_SMB )
  {
    rc = smb_create ( NULL,
      SMB_LOCAL | SMB_ZERO,
      nbytes,
      0,
      0,
      0,
      ( SMB_handle *) ph_mem_info );
    if ( rc )
    {
      printf ( "\nError: %s:%d : smb_create()\n", __FILE__, __LINE__ );
      err_print ( rc, ERR_GET_ALL );
      return ( NULL );
    }
    rc = smb_map ( * ( ( SMB_handle *) ( ph_mem_info ) ),
      0, 0, 0, 0, 0, 0,
      &hmap,
      &hmap_info );
    if ( rc )
    {
      printf ( "\nError: %s:%d : smb_map()\n", __FILE__, __LINE__ );
      err_print ( rc, ERR_GET_ALL );
      return ( NULL );
    }
      return ( hmap_info.process.vaddr );
    }
    else
  #else
    ( void ) flag;
  #endif
    {
      return ( malloc ( nbytes ) );
    }
  }
  /** -- Free memory with an SMB or malloc () -- **/
  /** p_mem must be the same pointer returned from mem_alloc().
  ph_mem_info must be the same pointer returned from mem_alloc(). **/
  void mem_free ( void *p_mem, SAL_i32 flag, void * ph_mem_info )
  {
  #if defined ( _MC_EXEC )
    SAL_i32 rc;
    if ( flag == MEM_FROM_SMB )
    {
      rc = smb_close ( * ( ( SMB_handle *) ( ph_mem_info ) ), 0 );
      if ( rc )
      {
        printf ( "\nError: %s:%d : smb_close()\n", __FILE__, __LINE__ );
        err_print ( rc, ERR_GET_ALL );
        return;
    }
    else
  #else
    ( void ) flag;
  #endif
    {
      if ( p_mem != NULL )
        free ( p_mem );
    }
    return;
    }

mem_utils.h

#ifndef MEM_UTILS_H
#define MEM_UTILS_H
/* ---- Memory utility function flags ---------------------------- */
#define MEM_FROM_HEAP ( 1 ) /* Allocate memory from heap */
#define MEM_FROM_SMB ( 2 ) /* Allocate memory from SMB (MCexec only) */
/* ---- Memory utility function prototypes ----------------------- */
void * mem_alloc ( SAL_ui32 n_bytes, SAL_i32 flag, void * ph_mem_info );
void mem_free ( void * p_mem, SAL_i32 flag, void * ph_mem_info );
#endif /* MEM_UTILS_H */
