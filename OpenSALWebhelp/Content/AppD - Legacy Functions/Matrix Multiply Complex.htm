Matrix Multiply

cmmulx(), cmmuldx(), zmmulx(), zmmuldx()

Synopsis

void cmmulx(
  SAL_cf32 *A, /* input matrix */
  SAL_i32 I, /* address stride for A, I=2i */
  SAL_cf32 *B, /* input matrix */
  SAL_i32 J, /* address stride for B, J=2j */
  SAL_cf32 *C, /* output matrix */
  SAL_i32 K, /* address stride for C, K=2k */
  SAL_i32 M, /* row count for A and C */
  SAL_i32 N, /* column count for B and C */
  SAL_i32 P, /* column count for A, row count for B */
  SAL_i32 flag ); /* ESAL flag */
void cmmuldx(
  SAL_cf64 *A, /* input matrix */
  SAL_i32 I, /* address stride for A, I=2i */
  SAL_cf64 *B, /* input matrix */
  SAL_i32 J, /* address stride for B, J=2j */
  SAL_cf64 *C, /* output matrix */
  SAL_i32 K, /* address stride for C, K=2k */
  SAL_i32 M, /* row count for A and C */
  SAL_i32 N, /* column count for B and C */
  SAL_i32 P, /* column count for A, row count for B */
  SAL_i32 flag ); /* ESAL flag */
void zmmulx(
  SAL_zf32 *A, /* input matrix */
  SAL_i32 i, /* address stride for A */
  SAL_zf32 *B, /* input matrix */
  SAL_i32 j, /* address stride for B */
  SAL_zf32 *C, /* output matrix */
  SAL_i32 k, /* address stride for C */
  SAL_i32 M, /* row count for A and C */
  SAL_i32 N, /* column count for B and C */
  SAL_i32 P, /* column count for A, row count for B */
  SAL_i32 flag ); /* ESAL flag */
void zmmuldx(
  SAL_zf64 *A, /* input matrix */
  SAL_i32 i, /* address stride for A */
  SAL_zf64 *B, /* input matrix */
  SAL_i32 j, /* address stride for B */
  SAL_zf64 *C, /* output matrix */
  SAL_i32 k, /* address stride for C */
  SAL_i32 M, /* row count for A and C */
  SAL_i32 N, /* column count for B and C */
  SAL_i32 P, /* column count for A, row count for B */
  SAL_i32 flag ); /* ESAL flag */

Description

The cmmulx(), cmmuldx(), zmmulx(), and zmmuldx() functions multiply M x P matrix A by P x N matrix B and store the results in M x N matrix C. Values for M, N, and P specify the number of complex elements in each row or column. These functions can be done out-of-place only.

Note 
The cmmulx(), cmmuldx(), zmmulx(), and zmmuldx() functions are not recommended for new application development. Use the newer cmat_mulx(), cmat_muldx(), zmat_mulx() and zmat_muldx() functions for matrix multiplication.
